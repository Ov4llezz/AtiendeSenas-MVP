# Aplicar fix de compatibilidad NumPy (necesario en Colab)
import sys
import numpy as np

# Monkey patch para compatibilidad NumPy 1.x <-> 2.x
class _CoreModule:
    def __getattr__(self, name):
        if hasattr(np.core, name):
            return getattr(np.core, name)
        if hasattr(np, name):
            return getattr(np, name)
        raise AttributeError(f"module 'numpy._core' has no attribute '{name}'")

sys.modules['numpy._core'] = _CoreModule()
sys.modules['numpy._core.multiarray'] = np.core.multiarray
sys.modules['numpy._core.umath'] = np.core.umath

print("✓ NumPy compatibility patch aplicado")
print(f"NumPy version: {np.__version__}")
